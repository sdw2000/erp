#### 步骤2: 改进分页响应处理（可选）
在 `loadScheduleList()` 中显式处理 `pages` 字段：

```javascript
async loadScheduleList() {
  // ... 现有代码 ...
  const res = await getScheduleList(params);
  if (res.code === 200) {
    this.scheduleList = res.data.list || [];
    this.total = res.data.total || 0;
    // 新增: 记录total pages信息（用于右下角显示）
    this.pages = res.data.pages || Math.ceil(this.total / this.queryParams.pageSize);
  }
}
```

#### 步骤3: 在UI中显示pages信息（可选）

```javascript
// 在data()中添加
coatingTasks: [],
coatingParams: {
  pageNum: 1,
  pageSize: 20,
  status: '',
  scheduleId: null
},
coatingTotal: 0,

// 在methods中添加
async loadCoatingTasks() {
  try {
    const res = await getCoatingTasks(this.coatingParams);
    if (res.code === 200) {
      this.coatingTasks = res.data.list || [];
      this.coatingTotal = res.data.total || 0;
    }
  } catch (error) {
    console.error('加载涂布任务失败:', error);
  }
},

handleCoatingPageChange(page) {
  this.coatingParams.pageNum = page;
  this.loadCoatingTasks();
},

handleCoatingSizeChange(size) {
  this.coatingParams.pageSize = size;
  this.coatingParams.pageNum = 1;
  this.loadCoatingTasks();
}
```

## 📝 API响应格式说明

后端所有分页API的标准响应格式：

```json
{
  "code": 200,
  "msg": "success",
  "data": {
    "list": [...],           // 当前页数据
    "total": 100,            // 总记录数
    "pageNum": 1,            // 当前页
    "pageSize": 10,          // 每页大小
    "pages": 10,             // ⭐ 总页数（新增）
    "hasNextPage": true      // 是否有下一页
  }
}
```
